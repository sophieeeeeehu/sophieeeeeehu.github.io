@typecheck
def read(filename: str) -> List[WidIndex]:
    """    
    reads information from the specified file and returns a list of WidIndex 
    from data that's reliable(not empty).
    """
    # return []  #stub
    # Template from HtDAP
    # lowi contains the result so far
    lowi = [] # type: List[WidIndex]

    with open(filename) as csvfile:
        
        reader = csv.reader(csvfile)
        next(reader) # skip header line

        for row in reader:
            # row[0] -> country name - str
            # row[2] -> bottom 50% income - str
            # row[3] -> top 10% income - str
            # row[4] -> top 1% income - str
            # row[5] -> bottom 50% wealth - str
            # row[6] -> top 10% wealth - str
            # row[6] -> top 1% wealth - str
            if is_reliable(row):
                wi = WidIndex(row[0], parse_float(row[2]), parse_float(row[3]),
                               parse_float(row[4]), parse_float(row[5]),
                               parse_float(row[6]), parse_float(row[7]))
                lowi.append(wi)
    
    return lowi

@typecheck
def is_reliable(row:List[str])->bool:
    '''
    Return True if the top 1% wealth(row[7]), top 10% wealth(row[6]) 
    and bottom 50% wealth(row[5]) is not an empty string,
    return False if not.
    ASSUMES: row is a full row of data from a WidIndex data information file. 
    Data except for row[5], row[6] and row[7] are all reliable(not empty).
    '''
    # return True # stub
    if row[5] == '' or row[6] == '' or row[7] == '':
        return False
    else:
        return True

@typecheck
def growth(income:float, wealth:float) -> float:
    '''
    gets a income and wealth and return the growth percentage from income and wealth to 1 decimal,
    For example:
    if income is 10 and wealth is 15, then the growth percentage will be 50% -> 50
    if income is 9 and wealth is 4, then the growth rate will be -55.55555% ->-55.6
    ASSUME: income is not 0
    '''
    # template based on atomic non-distinct
    return round((wealth-income)/income * 100,1)

@typecheck
def get_income_share(wi: WidIndex, percentage:int)->float:
    '''
    gets a wid index and a specific income percentage group, 
    where the percentage can be 1(top 1%), 10(top 10%) or 50(bottom 50%) in int.
    returns the income share of the group of the given percentage.
    '''
    # template based on Wid Index
    # return 0 # stub
    
    if percentage == 1:
        return wi.top_1_income
    elif percentage == 10:
        return wi.top_10_income
    elif percentage == 50:
        return wi.bottom_50_income

@typecheck
def get_wealth_share(wi: WidIndex, percentage:int)->float:
    '''
    gets a wid index and a specific wealth percentage group, 
    where the percentage can be 1(top 1%), 10(top 10%) or 50(bottom 50%) in int.
    returns the wealth share of the group of the given percentage.
    '''
    # template based on Wid Index
    # return 0 # stub
    
    if percentage == 1:
        return wi.top_1_wealth
    elif percentage == 10:
        return wi.top_10_wealth
    elif percentage == 50:
        return wi.bottom_50_wealth

@typecheck
def growth_of_group(lowi:List[WidIndex], percentage: int)->List[float]:
    '''
    gets a list of WidIndex and returns a list of float of the growth percentage from income and wealth 
    of bottom given income percentage group.
    the percentage can be 1, 10 or 50(int)
    1 represents the top 1% income group
    10 represents the top 10% income group
    50 represents the bottom 50% income group
    '''
    # template from List[WidIndex]
    # return [] # stub
    
    # the current list of float of the growth percentage from income and wealth of the given group.
    current_list = [] # type: List[float]
    for wi in lowi:
        current_list.append(growth(get_income_share(wi, percentage), get_wealth_share(wi, percentage)))
    return current_list
